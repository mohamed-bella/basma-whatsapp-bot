API Endpoints
All requests need header: x-api-key: basma_api_secret_2024
Method Endpoint Use Case
POST /api/send-message Send any custom text to a phone number
POST /api/send-order-message Send a formatted order confirmation
GET /api/health Check if bot is running & connected
📋 Body fields for /api/send-message:
phone required — Phone with country code, no + (e.g. 212704969534)
message required — The text to send. Supports WhatsApp bold (*text*) and newlines.

📋 Body fields for /api/send-order-message:
phone, name, order_id, product, price required · date optional

// ── Send any message when a form is submitted ──────────────────
async function sendWhatsAppMessage(phone, message) {
  const response = await fetch('https://seven-rice-try.loca.lt/api/send-message', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': 'basma_api_secret_2024'      // from your .env API_SECRET
    },
    body: JSON.stringify({ phone, message })
  });

  const data = await response.json();
  return data;
}

// ── Example 1: Contact form submitted ──────────────────────────
document.getElementById('contact-form').addEventListener('submit', async (e) => {
  e.preventDefault();
  const phone = document.getElementById('phone').value;    // e.g. "212704969534"
  const name  = document.getElementById('name').value;

  await sendWhatsAppMessage(phone,
    `👋 Hello ${name}!\n\nThank you for contacting Basma Travel.\nOur team will call you shortly. 🌍`
  );
});

// ── Example 2: Order placed on your website ────────────────────
async function notifyOrderConfirmed(order) {
  await fetch('https://seven-rice-try.loca.lt/api/send-order-message', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': 'basma_api_secret_2024'
    },
    body: JSON.stringify({
      phone:    order.clientPhone,     // "212704969534"
      name:     order.clientName,      // "Ahmed"
      order_id: order.id,              // "1254"
      product:  order.tourName,        // "Sahara Desert Tour"
      price:    order.price,           // "200€"
      date:     order.tourDate         // "12 April 2025" (optional)
    })
  });
}

// ── Send WhatsApp message from PHP (e.g. WooCommerce, Laravel) ─
function sendWhatsAppMessage($phone, $message) {
    $url  = 'https://seven-rice-try.loca.lt/api/send-message';
    $data = ['phone' => $phone, 'message' => $message];

    $ch = curl_init($url);
    curl_setopt_array($ch, [
        CURLOPT_POST           => true,
        CURLOPT_POSTFIELDS     => json_encode($data),
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_HTTPHEADER     => [
            'Content-Type: application/json',
            'x-api-key: basma_api_secret_2024'
        ]
    ]);

    $response = json_decode(curl_exec($ch), true);
    curl_close($ch);
    return $response;
}

// ── Usage: after WooCommerce order ─────────────────────────────
add_action('woocommerce_payment_complete', function($order_id) {
    $order = wc_get_order($order_id);
    $phone = $order->get_billing_phone();  // e.g. "+212704969534"
    $name  = $order->get_billing_first_name();

    sendWhatsAppMessage($phone,
        "✅ Bonjour {$name}!\n\nVotre commande #{$order_id} est confirmée.\nNous vous contacterons bientôt. 🌍"
    );
});
